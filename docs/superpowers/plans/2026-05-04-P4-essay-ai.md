**Goal:** 实现申论题库浏览、在线练习、AI 流式批改（评分+采分点+建议）、行测 AI 智能分析（诊断报告）、题目 AI 解析（参考答案+解题步骤），前端可完成写申论→AI批改→看评分、做行测→AI分析→看诊断的完整闭环。

**Tech Stack:** FastAPI 0.115.12, uvicorn 0.34.2, httpx 0.28.1, python-jose 3.3.0, sse-starlette 2.2.1, pydantic 2.10.6, Spring Boot 3.5.12, Vue 3, VitePress 2.0.0-alpha.15

```python
# ai-service/app/core/config.py
import os
import yaml
from pathlib import Path
from pydantic import BaseModel
from typing import Optional


class LLMProvider(BaseModel):
    api_url: str
    api_key: str
    model: str
    max_tokens: int = 4096


class LLMConfig(BaseModel):
    default_provider: str = "openai-compatible"
    providers: dict[str, LLMProvider] = {}


class JWTConfig(BaseModel):
    secret: str = "your-jwt-secret-same-as-java"
    algorithm: str = "HS256"


class AppConfig(BaseModel):
    llm: LLMConfig = LLMConfig()
    jwt: JWTConfig = JWTConfig()


def load_config() -> AppConfig:
    config_path = Path(__file__).parent.parent.parent / "config.yaml"
    data = {}
    if config_path.exists():
        with open(config_path, "r", encoding="utf-8") as f:
            data = yaml.safe_load(f) or {}

    cfg = AppConfig(**data)

    # 环境变量覆盖
    env_provider = os.getenv("LLM_PROVIDER")
    if env_provider:
        cfg.llm.default_provider = env_provider

    provider = cfg.llm.providers.get(cfg.llm.default_provider)
    if provider:
        env_url = os.getenv("LLM_API_URL")
        if env_url:
            provider.api_url = env_url
        env_key = os.getenv("LLM_API_KEY")
        if env_key:
            provider.api_key = env_key
        env_model = os.getenv("LLM_MODEL")
        if env_model:
            provider.model = env_model

    env_jwt_secret = os.getenv("JWT_SECRET")
    if env_jwt_secret:
        cfg.jwt.secret = env_jwt_secret

    return cfg


app_config = load_config()
```

```python
# ai-service/app/core/auth.py
from fastapi import Request, HTTPException
from jose import jwt, JWTError, ExpiredSignatureError
from .config import app_config


def get_current_user_id(request: Request) -> int:
    """从 Authorization header 提取 user_id，验证失败抛 401"""
    auth = request.headers.get("Authorization", "")
    if not auth.startswith("Bearer "):
        raise HTTPException(status_code=401, detail="未登录")

    token = auth[7:]
    try:
        payload = jwt.decode(
            token,
            app_config.jwt.secret,
            algorithms=[app_config.jwt.algorithm]
        )
        user_id = payload.get("user_id") or payload.get("sub")
        if user_id is None:
            raise HTTPException(status_code=401, detail="Token 无效")
        return int(user_id)
    except ExpiredSignatureError:
        raise HTTPException(status_code=401, detail="Token 已过期")
    except JWTError:
        raise HTTPException(status_code=401, detail="Token 无效")
```

```python
# ai-service/app/services/llm_service.py
import json
import httpx
from typing import AsyncIterator
from ..core.config import app_config


class LLMService:
    """统一 LLM 调用层，支持 OpenAI-compatible 和 Anthropic-compatible 格式"""

    def __init__(self):
        self.config = app_config.llm

    def _get_provider(self) -> dict:
        provider_name = self.config.default_provider
        provider = self.config.providers.get(provider_name)
        if not provider:
            raise ValueError(f"LLM provider '{provider_name}' not configured")
        return provider_name, provider

    async def chat_stream(self, messages: list[dict], max_tokens: int = None) -> AsyncIterator[str]:
        """
        流式调用 LLM，yield 每个 chunk 的文本内容。
        自动根据 provider 格式选择 OpenAI 或 Anthropic 协议。
        """
        provider_name, provider = self._get_provider()
        tok_limit = max_tokens or provider.max_tokens

        if "openai" in provider_name:
            async for chunk in self._stream_openai(provider, messages, tok_limit):
                yield chunk
        elif "anthropic" in provider_name:
            async for chunk in self._stream_anthropic(provider, messages, tok_limit):
                yield chunk
        else:
            # 默认走 OpenAI 格式
            async for chunk in self._stream_openai(provider, messages, tok_limit):
                yield chunk

    async def _stream_openai(self, provider, messages: list, max_tokens: int) -> AsyncIterator[str]:
        """OpenAI-compatible SSE 格式"""
        headers = {
            "Authorization": f"Bearer {provider.api_key}",
            "Content-Type": "application/json",
        }
        payload = {
            "model": provider.model,
            "messages": messages,
            "max_tokens": max_tokens,
            "stream": True,
        }

        async with httpx.AsyncClient(timeout=120.0) as client:
            async with client.stream("POST", provider.api_url, json=payload, headers=headers) as resp:
                resp.raise_for_status()
                async for line in resp.aiter_lines():
                    if not line.startswith("data: "):
                        continue
                    data = line[6:]
                    if data.strip() == "[DONE]":
                        break
                    try:
                        obj = json.loads(data)
                        delta = obj.get("choices", [{}])[0].get("delta", {})
                        content = delta.get("content", "")
                        if content:
                            yield content
                    except json.JSONDecodeError:
                        continue

    async def _stream_anthropic(self, provider, messages: list, max_tokens: int) -> AsyncIterator[str]:
        """Anthropic-compatible SSE 格式"""
        headers = {
            "x-api-key": provider.api_key,
            "anthropic-version": "2023-06-01",
            "Content-Type": "application/json",
        }

        # 分离 system prompt
        system_msg = ""
        user_messages = []
        for m in messages:
            if m["role"] == "system":
                system_msg += m["content"] + "\n"
            else:
                user_messages.append(m)

        payload = {
            "model": provider.model,
            "max_tokens": max_tokens,
            "stream": True,
        }
        if system_msg:
            payload["system"] = system_msg.strip()
        payload["messages"] = user_messages

        async with httpx.AsyncClient(timeout=120.0) as client:
            async with client.stream("POST", provider.api_url, json=payload, headers=headers) as resp:
                resp.raise_for_status()
                async for line in resp.aiter_lines():
                    if not line.startswith("data: "):
                        continue
                    data = line[6:]
                    try:
                        obj = json.loads(data)
                        if obj.get("type") == "content_block_delta":
                            text = obj.get("delta", {}).get("text", "")
                            if text:
                                yield text
                    except json.JSONDecodeError:
                        continue


llm_service = LLMService()
```

```java
            entity.setPaperId(paperId);
            entity.setQuestionId(questionId);
            entity.setSortOrder(sortOrder);
            entity.setUserAnswer(userAnswer);

            if (data.get("total_score") != null)
                entity.setTotalScore(new java.math.BigDecimal(data.get("total_score").toString()));
            if (data.get("max_score") != null)
                entity.setMaxScore(new java.math.BigDecimal(data.get("max_score").toString()));
            if (data.get("base_score") != null)
                entity.setBaseScore(new java.math.BigDecimal(data.get("base_score").toString()));
            if (data.get("base_max_score") != null)
                entity.setBaseMaxScore(new java.math.BigDecimal(data.get("base_max_score").toString()));
            if (data.get("content_score") != null)
                entity.setContentScore(new java.math.BigDecimal(data.get("content_score").toString()));
            if (data.get("content_max_score") != null)
                entity.setContentMaxScore(new java.math.BigDecimal(data.get("content_max_score").toString()));
            if (data.get("expression_score") != null)
                entity.setExpressionScore(new java.math.BigDecimal(data.get("expression_score").toString()));
            if (data.get("expression_max_score") != null)

            entity.setProblems(om.writeValueAsString(data.getOrDefault("problems", List.of())));
```

```javascript
export const aiApi = {
  // 申论批改（SSE 流式）
  essayReviewStream(data, onChunk, onResult, onError, onDone) {
    return ssePost(`${AI_BASE}/essay-review/stream`, data, onChunk, onResult, onError, onDone)
  },

  // 行测分析（SSE 流式）
  analysisStream(data, onChunk, onResult, onError, onDone) {
    return ssePost(`${AI_BASE}/analysis/stream`, data, onChunk, onResult, onError, onDone)
  },

  // 题目解析（SSE 流式）
  questionAnalysisStream(data, onChunk, onResult, onError, onDone) {
    return ssePost(`${AI_BASE}/question/analysis`, data, onChunk, onResult, onError, onDone)
  },

  // 剩余批改次数
  getRemainingCount() {
    return get(`${AI_BASE}/essay-review/count/remaining`)
  },

// SSE POST 辅助函数
function ssePost(url, data, onChunk, onResult, onError, onDone) {
  const token = localStorage.getItem('access_token')
  const fullUrl = url.startsWith('http') ? url : window.location.origin + url

  fetch(fullUrl, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${token}`,
    },
    body: JSON.stringify(data),
  }).then(response => {
    if (!response.ok) throw new Error(`HTTP ${response.status}`)
    const reader = response.body.getReader()
    const decoder = new TextDecoder()
    let buffer = ''

    function read() {
      reader.read().then(({ done, value }) => {
        if (done) { onDone && onDone(); return }
        buffer += decoder.decode(value, { stream: true })

        const lines = buffer.split('\n')
        buffer = lines.pop() || ''

        for (const line of lines) {
          if (line.startsWith('event: ')) {
            const eventType = line.slice(7).trim()
            // 下一行是 data
            continue
          }
          if (line.startsWith('data: ')) {
            const payload = line.slice(6)
            // 根据最近的 event 类型分发（简化：直接用内容判断）
            if (payload === '') continue
            try {
              // 尝试解析为 JSON（result 事件）
              const json = JSON.parse(payload)
              onResult && onResult(json)
            } catch {
              // 纯文本 chunk
              onChunk && onChunk(payload)
            }
          }
        }
        read()
      }).catch(err => onError && onError(err))
    }
    read()
  }).catch(err => onError && onError(err))
}
```
